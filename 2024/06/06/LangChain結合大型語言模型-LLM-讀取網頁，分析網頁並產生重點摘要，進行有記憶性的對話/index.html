е®ҳз¶Іж–ҮжӘ”

иҖҢжҜ”ијғеёёеҮәзҸҫзҡ„зҜ„дҫӢпјҢеҸҜд»ҘеҸғиҖғе®ҳз¶Іж–ҮжӘ”гҖӮ

еғҸжҳҜеҰӮжһңдҪ йңҖиҰҒCSVLoaderпјҢе°ұеҸҜд»Ҙз…§и‘—е®ҳж–№зөҰзҡ„зҜ„дҫӢдҝ®ж”№гҖӮ

from langchain_community.document_loaders.csv_loader import CSVLoader

жң¬ж–ҮйҖҷж¬ЎжңғдҪҝз”ЁWebBaseLoaderдҫҶеҠ ијүwebз¶Ій ҒпјҢеҸҜд»ҘзңӢеҲ°еҸійӮҠзҡ„д»Ӣзҙ№иӘӘйҖҷеҖӢеҘ—д»¶жҳҜдҪҝз”ЁBeautifulSoupдҫҶеҜҰдҪңзҡ„гҖӮ

еҜҰдҪңж•ҷеӯё

1.еҠ ијүе’Ңиҷ•зҗҶж–ҮжӘ”

жҲ‘еҖ‘е°ҮдҪҝз”Ё WebBaseLoader еҫһжҢҮе®ҡзҡ„з¶ІеқҖеҠ ијүж–ҮжӘ”

from langchain_community.document_loaders import WebBaseLoader

# дҪҝз”Ё WebBaseLoader еҫһжҢҮе®ҡз¶ІеқҖеҠ ијүж–ҮжӘ”
loader = WebBaseLoader("https://www.ollama.com/")
docs = loader.load()

жң¬ж–ҮдҪҝз”Ёhttps://www.ollama.com дҫҶеҜҰдҪңпјҢеӣ зӮәйҖҷеҖӢз¶Ій ҒжқұиҘҝжҜ”ијғе°‘пјҢжүҖд»Ҙи·‘жҜ”ијғеҝ«гҖӮ

2.TextSplitterеҲҮеүІжҲҗchunk

дҪҝз”Ё RecursiveCharacterTextSplitter е°Үж–ҮжӘ”еҲҶеүІжҲҗијғе°Ҹзҡ„еЎҠгҖӮ

from langchain_text_splitters import RecursiveCharacterTextSplitter

# дҪҝз”Ё RecursiveCharacterTextSplitter е°Үж–ҮжӘ”еҲҶеүІжҲҗијғе°Ҹзҡ„еЎҠ
text_splitter = RecursiveCharacterTextSplitter(chunk_size=1000, chunk_overlap=100)
documents = text_splitter.split_documents(docs)

3.еҲқе§ӢеҢ–иӘһиЁҖжЁЎеһӢе’ҢеөҢе…ҘжЁЎеһӢ

еҲқе§ӢеҢ– Ollama иӘһиЁҖжЁЎеһӢе’ҢеөҢе…ҘжЁЎеһӢпјҢдёҰдҪҝз”Ё FAISS еүөе»әеҗ‘йҮҸж•ёж“ҡеә«дҫҶеӯҳе„ІеҲҶеүІеҫҢзҡ„ж–ҮжӘ”гҖӮ

from langchain_community.llms import Ollama
from langchain_community.embeddings import OllamaEmbeddings
from langchain_community.vectorstores import FAISS

# еҲқе§ӢеҢ– Ollama иӘһиЁҖжЁЎеһӢ
llm = Ollama(model='llama3')
# еҲқе§ӢеҢ– Ollama еөҢе…ҘжЁЎеһӢ
embeddings = OllamaEmbeddings()

# дҪҝз”Ё FAISS еүөе»әеҗ‘йҮҸж•ёж“ҡеә«дёҰеҠ ијүеҲҶеүІеҫҢзҡ„ж–ҮжӘ”
vector = FAISS.from_documents(documents, embeddings)
retriever = vector.as_retriever()

4.еүөе»әжҸҗзӨәжЁЎжқҝ

жҲ‘еҖ‘йңҖиҰҒе…©еҖӢжҸҗзӨәжЁЎжқҝпјҡдёҖеҖӢз”Ёж–јз”ҹжҲҗжҗңзҙўжҹҘи©ўпјҢеҸҰдёҖеҖӢз”Ёж–јеҹәж–јжӘўзҙўеҲ°зҡ„ж–ҮжӘ”з”ҹжҲҗзӯ”жЎҲгҖӮ

from langchain_core.prompts import ChatPromptTemplate
from langchain_core.prompts import MessagesPlaceholder

# еүөе»әз”Ёж–јз”ҹжҲҗжҗңзҙўжҹҘи©ўзҡ„жҸҗзӨәжЁЎжқҝ
prompt_search_query = ChatPromptTemplate.from_messages([
    MessagesPlaceholder(variable_name="chat_history"),
    ("user", "{input}"),
    ("user", "Given the above conversation, generate a search query to look up in order to get information relevant to the conversation")
])

# еүөе»әз”Ёж–јзҚІеҸ–зӯ”жЎҲзҡ„жҸҗзӨәжЁЎжқҝ
prompt_get_answer = ChatPromptTemplate.from_messages([
    ('system', 'Answer the user\'s questions based on the below context:\n\n{context}'),
    MessagesPlaceholder(variable_name="chat_history"),
    ('user', '{input}'),
])

5.еүөе»әжӘўзҙўйҸҲе’Ңж–ҮжӘ”йҸҲ

жҲ‘еҖ‘дҪҝз”ЁдёҠиҝ°жҸҗзӨәжЁЎжқҝеүөе»әеё¶жңүжӯ·еҸІдёҠдёӢж–Үзҡ„жӘўзҙўеҷЁе’Ңж–ҮжӘ”йҸҲпјҢз„¶еҫҢзөҗеҗҲйҖҷе…©иҖ…дҫҶеүөе»әжӘўзҙўйҸҲгҖӮ

from langchain.chains import create_history_aware_retriever, create_retrieval_chain
from langchain.chains.combine_documents import create_stuff_documents_chain

# еүөе»әеё¶жңүжӯ·еҸІдёҠдёӢж–Үзҡ„жӘўзҙўеҷЁ
retriever_chain = create_history_aware_retriever(llm, retriever, prompt_search_query)

# еүөе»әж–ҮжӘ”йҸҲд»Ҙз”ҹжҲҗзӯ”жЎҲ
document_chain = create_stuff_documents_chain(llm, prompt_get_answer)

# зөҗеҗҲжӘўзҙўеҷЁе’Ңж–ҮжӘ”йҸҲпјҢеүөе»әжӘўзҙўйҸҲ
retrieval_chain_combine = create_retrieval_chain(retriever_chain, document_chain)

6.ж§Ӣе»әе•Ҹзӯ”зі»зөұ

жңҖеҫҢпјҢжҲ‘еҖ‘ж§Ӣе»әдёҖеҖӢеҫӘз’°дҫҶжҺҘж”¶з”ЁжҲ¶ијёе…ҘпјҢиӘҝз”ЁжӘўзҙўйҸҲз”ҹжҲҗеӣһзӯ”пјҢдёҰprintзөҗжһңгҖӮ

from langchain_core.messages import HumanMessage, AIMessage

# еҲқе§ӢеҢ–иҒҠеӨ©жӯ·еҸІиЁҳйҢ„
chat_history = []
input_text = input('>>> ')
while input_text.lower() != 'bye':
    if input_text:
        # иӘҝз”ЁжӘўзҙўйҸҲдёҰзҚІеҸ–еӣһзӯ”
        response = retrieval_chain_combine.invoke({
            'input': input_text,
            'chat_history': chat_history,
        })
        # жү“еҚ°еӣһзӯ”
        print(response['answer'])
        # е°Үз”ЁжҲ¶ијёе…Ҙе’Ң AI еӣһзӯ”ж·»еҠ еҲ°иҒҠеӨ©жӯ·еҸІдёӯ
        chat_history.append(HumanMessage(content=input_text))
        chat_history.append(AIMessage(content=response['answer']))

    input_text = input('>>> ')
